class LLMEngine: def __init__(self, model, **kwargs): # 1. 配置解析 config = Config(model, **config_kwargs) # 2. 启动多进程 (Tensor Parallelism) for i in range(1, config.tensor_parallel_size): process = ctx.Process(target=ModelRunner, args=(config, i, event)) process.start() # 3. 初始化 ModelRunner self.model_runner = ModelRunner(config, 0, events) self.scheduler = Scheduler(config) def generate(self, prompts, sampling_params): for prompt, sp in zip(prompts, sampling_params): self.add_request(prompt, sp) # 主循环: 持续调度直到全部完成 while not self.is_finished(): output, num_tokens = self.step() # 处理输出... return outputs def step(self): # 调度 → 模型运行 → 后处理 seqs, is_prefill = self.scheduler.schedule() token_ids = self.model_runner.call("run", seqs, is_prefill) self.scheduler.postprocess(seqs, token_ids) return outputs, num_tokens

📋 Scheduler 调度策略

Prefill-Decode 分离的连续批处理

def schedule(self) -> tuple[list[Sequence], bool]: # ====== PREPHASE (is_prefill=True) ====== scheduled_seqs = [] while self.waiting and num_seqs < self.max_num_seqs: seq = self.waiting[0] # 检查资源是否足够 if num_batched_tokens + len(seq) > max_batched_tokens: break if not self.block_manager.can_allocate(seq): break # 分配 block，状态变为 RUNNING self.block_manager.allocate(seq) seq.status = SequenceStatus.RUNNING self.waiting.popleft() self.running.append(seq) scheduled_seqs.append(seq) if scheduled_seqs: return scheduled_seqs, True # ====== DECODE PHASE (is_prefill=False) ====== while self.running and num_seqs < self.max_num_seqs: seq = self.running.popleft() # 检查是否可以追加 token while not self.block_manager.can_append(seq): if self.running: self.preempt(self.running.pop()) # 抢占 else: self.preempt(seq) break self.block_manager.may_append(seq) scheduled_seqs.append(seq) return scheduled_seqs, False

💾 BlockManager 内存管理

PagedAttention + Prefix Caching 实现

📑 分页管理

KV Cache 按固定大小 (256 tokens/block) 分页，避免 PyTorch 动态切片的内存碎片

🔍 Prefix 缓存

使用 xxhash 计算 block hash，相同前缀的序列可复用已计算的 KV

# Block 分配与 Prefix Caching def allocate(self, seq): for i in range(seq.num_blocks): token_ids = seq.block(i) # 计算链式 hash h = self.compute_hash(token_ids, prefix=h) # 查找缓存 block_id = self.hash_to_block_id.get(h, -1) if block_id == -1: # Cache miss: 分配新 block block_id = self.free_block_ids[0] self._allocate_block(block_id) else: # Cache hit: 复用 seq.num_cached_tokens += self.block_size self.blocks[block_id].ref_count += 1 seq.block_table.append(block_id) def compute_hash(cls, token_ids, prefix=-1): h = xxhash.xxh64() if prefix != -1: h.update(prefix.to_bytes(8, "little")) h.update(np.array(token_ids).tobytes()) return h.intdigest()

🔃 完整数据流

从请求到输出的完整路径

  1. Client: llm.generate(prompts, sampling_params)
  2. Tokenizer: 文本 → token IDs
  3. Sequence: 创建序列对象，状态=WAITING
  4. Scheduler.add(): 加入 waiting 队列
  5. Scheduler.schedule():
    • Prefill: 分配 blocks，计算完整 KV Cache
    • Decode: 追加单个 token，更新 block
  6. ModelRunner.run():
    • prepare_prefill() / prepare_decode()
    • model.forward() → compute_logits()
  7. Attention: FlashAttention + 写入 KV Cache
  8. Sampler: 采样生成新 token
  9. Scheduler.postprocess():
    • append_token() - 追加到序列
    • 检查结束条件 (EOS / max_tokens)
    • 释放 blocks
  10. Tokenizer: token IDs → 文本
  11. Return: [{"text": "...", "token_ids": [...]}]
